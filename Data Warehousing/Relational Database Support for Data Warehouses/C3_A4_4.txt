SELECT CALQUARTER,
ZIP,
NAME,
SUM(TOTALEXTCOST) AS TOTALEXTCOST,
SUM(TRANSACTIONCOUNT) AS TRANSACTIONCOUNT
FROM
(SELECT D.CALQUARTER AS CALQUARTER,
CV.ZIP AS ZIP,
CV.NAME AS NAME,
SVM.TOTALEXTCOST AS TOTALEXTCOST,
SVM.TRANSACTIONCOUNT AS TRANSACTIONCOUNT
FROM SALESBYVENDORDATEKEYMV2011 SVM
JOIN DATE_DIM D
ON SVM.DATEKEY = D.DATEKEY
JOIN CUST_VENDOR_DIM CV
ON SVM.CUSTVENDORKEY = CV.CUSTVENDORKEY
